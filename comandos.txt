Contar escolas: 

DELIMITER $$	
CREATE FUNCTION contarEscolas()
RETURNS INT
DETERMINISTIC
BEGIN
    DECLARE quantidade INT;

    SELECT COUNT(*) INTO quantidade FROM escola_campus;

    RETURN quantidade;
END $$
